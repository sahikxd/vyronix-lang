Explore the power of VYRONIX through these snippets.

1. Fibonacci Sequence

Classic recursive implementation showing function calls and basic math.

fn fib(n: i64) -> i64 {
    if (n <= 1) return n;
    return fib(n - 1) + fib(n - 2);
}

io.print(fib(10)); // Output: 55

2. Higher-Order Functions & Closures

Functions that return functions, capturing local state.

fn multiplier(factor: i64) -> fn(i64) -> i64 {
    return fn(n: i64) { 
        return n * factor; 
    };
}

let double = multiplier(2);
io.print(double(5)); // Output: 10

3. Structs & Data Modeling

Defining custom types with fields.

struct Point {
    x: f64,
    y: f64
}

let p = Point { x: 10.5, y: 20.0 };
io.print(p.x);

4. Error Handling (Try/Catch)

Safe execution and exception management.

try {
    throw "Something went wrong!";
} catch (e) {
    io.print("Caught error: " + e);
}

5. Generics

Writing reusable code for any data type.

fn identity(value: T) -> T {
    return value;
}

let a = identity(42);
let b = identity("Hello");
